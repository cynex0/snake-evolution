110 private static ArrayList<Player> readFromFile() {
111 JSONParser parser = new JSONParser();
112 ArrayList<Player> players = new ArrayList<>();
113 File file = new File("assets/Top10Scores.json");
114
115 try {
116 if (!file.exists()) {
117 JSONObject jsonObj = new JSONObject();
118 FileWriter writer = new FileWriter(file);
119 writer.write(jsonObj.toJSONString());
120 writer.close();
121 } else {
122 FileReader reader = new FileReader(file); // Creating reader to read data from json file
123 JSONObject readJsonObj = (JSONObject) parser.parse(reader); // parsing data from json file to string
124
125 for (Object PlayersData : readJsonObj.keySet()) {
126 String playerName = (String) PlayersData;
127 long playerScore = (long) readJsonObj.get(PlayersData);
128 Player player = new Player(playerName, playerScore);
129 players.add(player);
130 }
131
132 // sorting players from high to low scores
133 Collections.sort(players);
134 }
135 }
136 catch (Exception e) {
137 throw new RuntimeException("Error while reading file!\n" + e.getMessage());
138 }
139
140 return players;
141 }

170 public static void createPlayer(String name, long score){

172
173 FileWriter writer;
174 JSONObject jsonObj = new JSONObject();
175
176 playerList.addAll(currentPlayers);
177 Player newPlayer = new Player(name, score);
178 playerList.add(newPlayer);
179
180 // Storing top 10 players' names and scores in json file
181 try {
182 writer = new FileWriter("assets/Top10Scores.json");
183 for (Player player : playerList) {
184 if (jsonObj.containsKey(player.getName()) && (long) jsonObj.get(player.getName()) > player.getScore()) {
185 continue; // do not put a lower score if an entry with the same name exists
186 }
187 jsonObj.put(player.getName(), player.getScore());
188 }
189 writer.write(jsonObj.toJSONString());
190 writer.close();
191
192 } catch (IOException e) {
193 throw new RuntimeException(e);
194 }
