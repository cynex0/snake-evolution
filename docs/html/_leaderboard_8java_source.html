111 private static ArrayList<Player> readFromFile() {
112 JSONParser parser = new JSONParser();
113 ArrayList<Player> players = new ArrayList<>();
114 File file = new File("res/Top10Scores.json");
115
116 try {
117 if (!file.exists()) {
118 JSONObject jsonObj = new JSONObject();
119 FileWriter writer = new FileWriter(file);
120 writer.write(jsonObj.toJSONString());
121 writer.close();
122 } else {
123 FileReader reader = new FileReader(file); // Creating reader to read data from json file
124 JSONObject readJsonObj = (JSONObject) parser.parse(reader); // parsing data from json file to string
125
126 for (Object PlayersData : readJsonObj.keySet()) {
127 String playerName = (String) PlayersData;
128 long playerScore = (long) readJsonObj.get(PlayersData);
129 Player player = new Player(playerName, playerScore);
130 players.add(player);
131 }
132
133 // sorting players from high to low scores
134 Collections.sort(players);
135 }
136 }
137 catch (Exception e) {
138 throw new RuntimeException("Error while reading file!\n" + e.getMessage());
139 }
140
141 return players;
142 }

171 public static void createPlayer(String name, long score){
172 ArrayList<Player> currentPlayers = readFromFile();
173
174 FileWriter writer;
175 JSONObject jsonObj = new JSONObject();
176
177 playerList.addAll(currentPlayers);
178 Player newPlayer = new Player(name, score);
179 playerList.add(newPlayer);
180
181 // Storing top 10 players' names and scores in json file
182 try {
183 writer = new FileWriter("res/Top10Scores.json");
184 for (Player player : playerList) {
185 if (jsonObj.containsKey(player.getName()) && (long) jsonObj.get(player.getName()) > player.getScore()) {
186 continue; // do not put a lower score if an entry with the same name exists
187 }
188 jsonObj.put(player.getName(), player.getScore());
189 }
190 writer.write(jsonObj.toJSONString());
191 writer.close();
192
193 } catch (IOException e) {
194 throw new RuntimeException(e);
195 }
196 }
