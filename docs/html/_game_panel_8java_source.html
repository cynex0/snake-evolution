139 private boolean doFoodCollisions() {
140 boolean eaten = false;
141 Iterator<Food> it = food.iterator();
142 ArrayList<Food> newFood = new ArrayList<>();
143
144 while (it.hasNext()) {
145 Food foodItem = it.next();
146 if (snake.checkCollisionWith(foodItem.getFoodLocation())) {
147 it.remove(); // remove consumed food from list
148 eaten = true;
149 applyFoodEffect(foodItem.getFoodType());
150
151 // always spawn new default food as soon as the previous one is eaten
152 if(foodItem.getFoodType() == FoodType.DEFAULT)
153 newFood.add(generateNewFoodItem(false));
154
155 // 33% to spawn new bonus food in a valid position, up to 1 normal and 2 bonus
156 if (food.size() < 2 && rand.nextFloat() <= 0.33)
157 newFood.add(generateNewFoodItem(true));
158
159 // spawn new obstacle every 5th time food is eaten
160 if (score % 5 == 0)
161 obstacles.add(new Obstacle(snake.getBody()));
162 }
163 }
164
165 food.addAll(newFood); // add all generated food from the temporary storage to the actual list
166 return eaten;
167 }

176 private void applyFoodEffect(FoodType foodType) {
177 switch (foodType) {
178 case DEFAULT -> {
179 score++;
180 }
181 case SPEEDFOOD -> {
182 fastMode = true;
183 adjustSnakeSpeed(2);
184 currentEffectLabel = "SPED UP!";
185 startTime = System.currentTimeMillis();
186 }
187 case SLOWFOOD -> {
188 slowMode = true;
189 adjustSnakeSpeed(0.5);
190 currentEffectLabel = "SLOWED!";
191 startTime = System.currentTimeMillis();
192 }
193 case PLUSFOOD -> {
194 score += 2;
195 }
196 case MINUSFOOD -> {
197 score -= 2;
198 if (score < 0) score = 0;
199 }
200 case CONTROLINVERTER -> {
201 keyInverter = true;
202 startTime = System.currentTimeMillis();
203 adjustSnakeSpeed(0.75);
204 currentEffectLabel = "CONFUSED!";
205 }
206 }
