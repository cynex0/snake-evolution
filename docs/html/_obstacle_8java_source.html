43 if (snakePos.contains(startPos)) return; // prevent spawning if started spawning inside of the snake

47 double growChance = 0.75; // chance for increasing the number of cells
48 // generate additional cells, up to MAX_SIZE, with diminishing probability
49 for (int i = 0; i < MAX_SIZE; i++) {
50 double roll = rand.nextDouble(); // random number to determine if the obstacle should grow more
51
52 if (roll <= growChance) {
53 // grow in a random direction
54 ArrayList<CellPosition> viableCells = new ArrayList<>();
55 CellPosition prevCell = cells.get(i);
56
57 if (prevCell.x + 1 <= GameConstants.MAX_CELL) // can grow to the right
58 viableCells.add(new CellPosition(prevCell.x + 1, prevCell.y));
59 if (prevCell.x - 1 >= GameConstants.MIN_CELL) // can grow to the left
60 viableCells.add(new CellPosition(prevCell.x - 1, prevCell.y));
61 if (prevCell.y + 1 <= GameConstants.MAX_CELL) // can grow down
62 viableCells.add(new CellPosition(prevCell.x, prevCell.y + 1));
63 if (prevCell.y - 1 >= GameConstants.MIN_CELL) // can grow up
64 viableCells.add(new CellPosition(prevCell.x, prevCell.y - 1));
65
66 if (viableCells.isEmpty()) // stop spawning if no viable cells found
67 break;
68

70 CellPosition nextCell = viableCells.get(rand.nextInt(viableCells.size()));
71 if (snakePos.contains(nextCell)) // stop spawning if spawned inside the snake
