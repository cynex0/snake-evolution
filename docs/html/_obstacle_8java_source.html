40 public void respawn(ArrayList<CellPosition> snakePos) {
41 CellPosition startPos = getRandomCell(); // gets a random cell to start the spawning process
42 if (snakePos.contains(startPos)) return; // prevent spawning if started spawning inside of the snake
43
44 cells.add(startPos); // add the starting cell to the list of obstacle cells
45
46 double growChance = 0.75; // chance for increasing the number of cells
47 // generate additional cells, up to MAX_SIZE, with diminishing probability
48 for (int i = 0; i < MAX_SIZE; i++) {
49 double roll = rand.nextDouble(); // random number to determine if the obstacle should grow more
50
51 if (roll <= growChance) {
52 // grow in a random direction
53 ArrayList<CellPosition> viableCells = new ArrayList<>();
54 CellPosition prevCell = cells.get(i);
55
56 if (prevCell.x + 1 <= GameConstants.MAX_CELL) // can grow to the right
57 viableCells.add(new CellPosition(prevCell.x + 1, prevCell.y));
58 if (prevCell.x - 1 >= GameConstants.MIN_CELL) // can grow to the left
59 viableCells.add(new CellPosition(prevCell.x - 1, prevCell.y));
60 if (prevCell.y + 1 <= GameConstants.MAX_CELL) // can grow down
61 viableCells.add(new CellPosition(prevCell.x, prevCell.y + 1));
62 if (prevCell.y - 1 >= GameConstants.MIN_CELL) // can grow up
63 viableCells.add(new CellPosition(prevCell.x, prevCell.y - 1));
64
65 if (viableCells.isEmpty()) // stop spawning if no viable cells found
66 break;
67
68 // choose a random cell out of the possible cells
69 CellPosition nextCell = viableCells.get(rand.nextInt(viableCells.size()));
70 if (snakePos.contains(nextCell)) // stop spawning if spawned inside the snake
71 break;
72
73 cells.add(nextCell);
74 growChance = growChance - 0.1; // add non-linearity to the chance
75 }
76 else break;
77 }
